function findDuplicates(arr) {
  return arr.filter((item, index) =>
    arr.indexOf(item) !== index);
}

// LOGIT explains what's actually happening ✦ LOGIT
→ indexOf() returns the FIRST occurrence.
→ filter() keeps items where position ≠ first occurrence.
→ Therefore it catches all duplicates. Try it with ["a","b","a","c","b"]
